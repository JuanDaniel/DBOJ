CREATE  VIEW  "public"."inciso_b"  (idconcierto,  nombre_concierto, 
lugar_presentacion, fecha, cantidad_publico, pago)
AS
SELECT conciertos.* FROM conciertos 
WHERE conciertos.idconcierto NOT IN (SELECT conciertos.idconcierto
FROM canciones JOIN canciones_conciertos 
ON canciones.idcancion = canciones_conciertos.idcancion
WHERE canciones.nombre_cancion = 'Me gusta tanto');
